25template <typename Resultant, typename... Parametric>
26struct Invocative {
27 using Interfacial = Resultant(const void* const, Parametric...);
28 Interfacial* interface;
29 const void* locality;
30 Resultant operator()(Parametric... arguments) const {
31 return interface(locality, arguments...);
32 }
33};
34template <typename Procedural, typename Resultant, typename... Parametric>
35Resultant
36InvokeProcedure(const void* locality, Parametric... arguments) {
37 using Classificatory = Procedural* const*;
38 const Classificatory objective = static_cast<Classificatory>(locality);
39 return (**objective)(arguments...);
40}
41template <typename Procedural, typename Resultant, typename... Parametric>
42static inline Invocative<Resultant, Parametric...>
43AssignInvokeProcedure(Procedural* const& locality) {
44 using Specific = Invocative<Resultant, Parametric...>;
45 static auto& Invoke = InvokeProcedure<Procedural, Resultant, Parametric...>;
46 const Specific invocation = { &Invoke, &locality };
47 return invocation;
48}

66struct lambda_callback_t {
67 using invoke_t = Invocative<CALLBACK_TYPE_LIST>;
68 using remover_t = void(void*);
69 invoke_t invoke;
70 void* lambda;
71 remover_t* remove;
72 lambda_callback_t() {
73 invoke.interface = 0;
74 invoke.locality = 0;
75 lambda = 0;
76 remove = 0;
77 }
78 lambda_callback_t(const lambda_callback_t& copy) {
79 invoke.interface = copy.invoke.interface;
80 invoke.locality = &lambda;
81 lambda = copy.lambda;
82 remove = copy.remove;
83 }
84 void replace(const lambda_callback_t& with) {
85 if (lambda)
86 remove(lambda);
87 invoke.interface = with.invoke.interface;
88 invoke.locality = &lambda;
89 lambda = with.lambda;
90 remove = with.remove;
91 }
92 void cleanup() {
93 if (lambda)
94 remove(lambda);
95 invoke.interface = 0;
96 invoke.locality = 0;
97 lambda = 0;
98 remove = 0;
99 }
100 };
101 template <typename LambdaTypical>
102 static inline lambda_callback_t
103 CallLambda(const LambdaTypical& lambda) {
104 static auto RemoveLambda = [](void* pointer) {
105 delete static_cast<LambdaTypical*>(pointer);
106 };
107 static auto& Assign = AssignInvokeProcedure<LambdaTypical, CALLBACK_TYPE_LIST>;
108 lambda_callback_t callback;
109 callback.lambda = new LambdaTypical(lambda);
110 callback.invoke = Assign(*reinterpret_cast<LambdaTypical**>(&callback.lambda));
111 callback.remove = RemoveLambda;
112 return callback; // Not intended for direct invocation, copying only
113 }
